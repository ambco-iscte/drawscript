N: 8
SIDE: 40
MARGIN: 5
WHITE: |255|
DARK_BLUE: |0|0|128|
BABY_BLUE: |137|207|240|
---
dimension: (N * SIDE + MARGIN * 2,N * SIDE + MARGIN * 2)
background: WHITE
---
setlinecolor DARK_BLUE
for l in [0, N[ {
  for c in [0, N[ {
    if ((l + c) % 2 == 0) {
      setfillcolor WHITE
    } else {
      setfillcolor BABY_BLUE
    }
    circle (c * SIDE + MARGIN,l * SIDE + MARGIN), SIDE
  }
}
